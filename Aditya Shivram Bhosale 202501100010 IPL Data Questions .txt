import pandas as pd
import numpy as np

# creating dataset
# took reference from kaggle ipl dataset for column names

np.random.seed(10)

teams = ['Mumbai Indians', 'Chennai Super Kings', 'Royal Challengers Bangalore',
         'Kolkata Knight Riders', 'Delhi Capitals', 'Sunrisers Hyderabad',
         'Rajasthan Royals', 'Punjab Kings']

players = {
    'Mumbai Indians': ['Rohit Sharma', 'Jasprit Bumrah', 'Hardik Pandya', 'Suryakumar Yadav', 'Kieron Pollard'],
    'Chennai Super Kings': ['MS Dhoni', 'Ravindra Jadeja', 'Faf du Plessis', 'Deepak Chahar', 'Ambati Rayudu'],
    'Royal Challengers Bangalore': ['Virat Kohli', 'AB de Villiers', 'Glenn Maxwell', 'Mohammed Siraj', 'Yuzvendra Chahal'],
    'Kolkata Knight Riders': ['Andre Russell', 'Sunil Narine', 'Eoin Morgan', 'Pat Cummins', 'Shubman Gill'],
    'Delhi Capitals': ['Rishabh Pant', 'Shikhar Dhawan', 'Kagiso Rabada', 'Anrich Nortje', 'Prithvi Shaw'],
    'Sunrisers Hyderabad': ['David Warner', 'Kane Williamson', 'Bhuvneshwar Kumar', 'Rashid Khan', 'Jonny Bairstow'],
    'Rajasthan Royals': ['Jos Buttler', 'Sanju Samson', 'Ben Stokes', 'Trent Boult', 'Yashasvi Jaiswal'],
    'Punjab Kings': ['KL Rahul', 'Chris Gayle', 'Mohammed Shami', 'Mayank Agarwal', 'Nicholas Pooran']
}

venues = ['Wankhede Stadium', 'MA Chidambaram Stadium', 'M Chinnaswamy Stadium',
          'Eden Gardens', 'Arun Jaitley Stadium', 'Rajiv Gandhi Stadium',
          'Sawai Mansingh Stadium', 'PCA Stadium']

seasons = list(range(2008, 2023))

rows = []
mid = 1
for season in seasons:
    num_matches = np.random.randint(56, 64)
    for _ in range(num_matches):
        t1 = np.random.choice(teams)
        t2 = np.random.choice(teams)
        while t1 == t2:
            t2 = np.random.choice(teams)

        s1 = np.random.randint(130, 220)
        s2 = np.random.randint(110, 215)
        winner = t1 if s1 > s2 else t2
        toss_w = np.random.choice([t1, t2])
        toss_d = np.random.choice(['bat', 'field'])
        pom = np.random.choice(players[winner])

        rows.append({
            'match_id': mid, 'season': season,
            'team1': t1, 'team2': t2,
            'venue': np.random.choice(venues),
            'toss_winner': toss_w, 'toss_decision': toss_d,
            'team1_score': s1, 'team2_score': s2,
            'winner': winner, 'win_margin': abs(s1 - s2),
            'player_of_match': pom
        })
        mid += 1

matches = pd.DataFrame(rows)

d_rows = []
for _, m in matches.iterrows():
    for inn in [1, 2]:
        bat_team = m['team1'] if inn == 1 else m['team2']
        bowl_team = m['team2'] if inn == 1 else m['team1']
        num_balls = np.random.randint(65, 121)

        for b in range(num_balls):
            runs = int(np.random.choice([0,1,2,3,4,6], p=[0.30,0.30,0.15,0.03,0.15,0.07]))
            extra = int(np.random.choice([0, 1], p=[0.93, 0.07]))
            wicket = int(np.random.choice([0, 1], p=[0.94, 0.06]))
            dkind = np.random.choice(['caught','bowled','lbw','run out','stumped']) if wicket else ''

            d_rows.append({
                'match_id': m['match_id'],
                'season': m['season'],
                'innings': inn,
                'over': b // 6,
                'ball': b % 6 + 1,
                'batting_team': bat_team,
                'bowling_team': bowl_team,
                'batsman': np.random.choice(players[bat_team]),
                'bowler': np.random.choice(players[bowl_team]),
                'batsman_runs': runs,
                'extra_runs': extra,
                'total_runs': runs + extra,
                'is_wicket': wicket,
                'dismissal_kind': dkind
            })

deliveries = pd.DataFrame(d_rows)

print("matches shape:", matches.shape)
print("deliveries shape:", deliveries.shape)
print()
print(matches.head(3))
print()

# Q11. What is the strike rate of top batsmen? (min 200 balls faced)
# method used - groupby, division, np.multiply

print("Q11) Strike rate of batsmen (min 200 balls faced)")
print()

balls_faced = deliveries.groupby('batsman')['ball'].count()
runs_made = deliveries.groupby('batsman')['batsman_runs'].sum()

# formula = (runs / balls) * 100
sr = np.multiply(runs_made / balls_faced, 100).round(2)
sr = sr[balls_faced >= 200].sort_values(ascending=False).head(10)
sr.name = 'strike_rate'
print(sr)
print()

# Q13. Which bowlers have the best economy rate? (min 300 balls)
# method used - groupby, divide, filter

print("Q13) Best economy rate - bowlers (min 300 balls bowled)")
print()

balls_bowled = deliveries.groupby('bowler')['ball'].count()
runs_given = deliveries.groupby('bowler')['total_runs'].sum()

economy = (runs_given / (balls_bowled / 6)).round(2)
economy = economy[balls_bowled >= 300].sort_values().head(8)
economy.name = 'economy_rate'
print(economy)
print()

# Q17. Compare average runs in powerplay vs death overs per match
# method used - filter by over number, groupby, np.mean

print("Q17) Powerplay vs Death over runs comparison")
print()

pp = deliveries[deliveries['over'] < 6]
death = deliveries[deliveries['over'] >= 16]

pp_runs = pp.groupby('match_id')['total_runs'].sum()
death_runs = death.groupby('match_id')['total_runs'].sum()

print("Avg Powerplay runs per match :", round(np.mean(pp_runs.values), 2))
print("Avg Death over runs per match:", round(np.mean(death_runs.values), 2))
print()

# Q20. Is there any correlation between winning toss and win margin?
# method used - np.corrcoef, pandas corr

print("Q20) Correlation between toss result and win margin")
print()

matches['toss_win_flag'] = (matches['toss_winner'] == matches['winner']).astype(int)

corr = np.corrcoef(matches['toss_win_flag'].values, matches['win_margin'].values)[0][1]
print("Correlation value (numpy):", round(corr, 4))

pd_corr = matches[['toss_win_flag', 'win_margin']].corr()
print()
print("Correlation matrix (pandas):")
print(pd_corr)
print()
print("Conclusion: value is very close to 0 so toss win has almost no effect on win margin")
